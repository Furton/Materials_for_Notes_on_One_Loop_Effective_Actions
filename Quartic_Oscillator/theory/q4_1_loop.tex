\noindent
The oscillator is
\begin{equation}
S[q]=\int \dd t\left[{m\over2}\dot q^2-{m\omega^2\over2}q^2-{\lambda\over24}q^4\right],
\qquad
S_E[q]=\int \dd\tau\left[{m\over2}\dot q^2+{m\omega^2\over2}q^2+{\lambda\over24}q^4\right].
\label{eq:actions}
\end{equation}
Dots denote derivatives with respect to the displayed time variable.  The two final results are Eq.~\eqref{eq:final-adiabatic}, the local adiabatic one-loop equation obtained by Wick rotating the Euclidean derivative expansion, and Eq.~\eqref{eq:final-nonlocal}, the causal one-loop equation with the first real-time memory correction obtained from the mode correlator.

\section{Euclidean expansion and the one-loop determinant}

Split
\begin{equation}
q(\tau)=Q(\tau)+\eta(\tau).
\label{eq:split}
\end{equation}
The kinetic term expands as
\begin{equation}
{m\over2}(\dot Q+\dot\eta)^2={m\over2}\dot Q^2+m\dot Q\dot\eta+{m\over2}\dot\eta^2,
\qquad
\int\dd\tau\,m\dot Q\dot\eta=-\int\dd\tau\,m\ddot Q\eta,
\label{eq:kin-expand}
\end{equation}
where boundary terms are assumed to vanish.  The potential expansion is
\begin{align}
{m\omega^2\over2}(Q+\eta)^2+{\lambda\over24}(Q+\eta)^4
={}&{m\omega^2\over2}Q^2+{\lambda\over24}Q^4
+\bigg(m\omega^2Q+{\lambda\over6}Q^3\bigg)\eta \nonumber \\
&+{1\over2}\bigg(m\omega^2+{\lambda\over2}Q^2\bigg)\eta^2
+{\lambda\over6}Q\eta^3+{\lambda\over24}\eta^4 .
\label{eq:pot-expand}
\end{align}
Therefore
\begin{equation}
S_E[Q+\eta]
=S_E[Q]+\int\dd\tau\,\eta E_Q
+{1\over2}\int\dd\tau\,\eta\mathcal O_Q\eta
+\int\dd\tau\left({\lambda\over6}Q\eta^3+{\lambda\over24}\eta^4\right),
\label{eq:expanded-action}
\end{equation}
with
\begin{equation}
E_Q=-m\ddot Q+m\omega^2Q+{\lambda\over6}Q^3,
\label{eq:EQ}
\end{equation}
\begin{equation}
\mathcal O_Q=-m{\dd^2\over\dd\tau^2}+m\omega^2+{\lambda\over2}Q^2(\tau)
=m\left[-{\dd^2\over\dd\tau^2}+\Om^2(\tau)\right],
\qquad
\Om^2(\tau)=\omega^2+{\lambda\over2m}Q^2(\tau).
\label{eq:operator}
\end{equation}
In the effective-action construction the source cancels the term linear in \(\eta\).  The Gaussian path integral then gives
\begin{equation}
\Gamma_E[Q]=S_E[Q]+{\hbar\over2}\Tr\ln\mathcal O_Q+\Oh(\hbar^2).
\label{eq:Gamma-one-loop}
\end{equation}
Since \(\Tr\ln\mathcal O_Q=\Tr\ln m+\Tr\ln[-\partial_\tau^2+\Om^2]\), the factor \(m\) contributes only a field-independent constant.  After also subtracting the \(Q=0\) determinant,
\begin{equation}
\Delta\Gamma_E^{(1)}
={\hbar\over2}\Tr\ln[-\partial_\tau^2+\Om^2]
-{\hbar\over2}\Tr\ln[-\partial_\tau^2+\omega^2].
\label{eq:det-subtracted}
\end{equation}

\section{Derivative expansion of the Euclidean determinant}

Let
\begin{equation}
A=-\partial_\tau^2+\Om^2(\tau),
\qquad
V(\tau)=\Om^2(\tau),
\qquad
f(\tau,p)=p^2+V(\tau).
\label{eq:AfV}
\end{equation}
The Weyl symbol \(R(\tau,p)\) of \(A^{-1}\) satisfies
\begin{equation}
f\star R=1,
\label{eq:star-inverse}
\end{equation}
where, to second adiabatic order,
\begin{equation}
f\star R=fR+{\ii\over2}(f_\tau R_p-f_pR_\tau)
-{1\over8}\left(f_{\tau\tau}R_{pp}-2f_{\tau p}R_{p\tau}+f_{pp}R_{\tau\tau}\right)+\Oh(\partial_\tau^4).
\label{eq:star-product}
\end{equation}
With \(R=R_0+R_2+\cdots\), Eq.~\eqref{eq:star-inverse} gives
\begin{equation}
R_0={1\over f}.
\label{eq:R0}
\end{equation}
The first-derivative term vanishes because
\begin{equation}
f_\tau(R_0)_p-f_p(R_0)_\tau=V'\left(-{2p\over f^2}\right)-2p\left(-{V'\over f^2}\right)=0.
\label{eq:first-derivative-cancel}
\end{equation}
Using \(f_{\tau p}=0\), \(f_{pp}=2\),
\begin{equation}
(R_0)_{pp}=-{2\over f^2}+{8p^2\over f^3},
\qquad
(R_0)_{\tau\tau}=-{V''\over f^2}+{2(V')^2\over f^3},
\label{eq:R0-derivatives}
\end{equation}
the second-order equation is
\begin{equation}
fR_2-{1\over8}\left[V''(R_0)_{pp}+2(R_0)_{\tau\tau}\right]=0.
\label{eq:R2-eq}
\end{equation}
Thus
\begin{equation}
R_2=-{V''\over2f^3}+{V''p^2\over f^4}+{(V')^2\over2f^4}.
\label{eq:R2}
\end{equation}
The diagonal Green function is \(G_E(\tau,\tau)=\int\frac{\dd p}{2\pi}R(\tau,p)\).  The needed integrals are
\begin{equation}
\int{\dd p\over2\pi}{1\over p^2+\Om^2}={1\over2\Om},
\quad
\int{\dd p\over2\pi}{1\over f^3}={3\over16\Om^5},
\quad
\int{\dd p\over2\pi}{p^2\over f^4}={1\over32\Om^5},
\quad
\int{\dd p\over2\pi}{1\over f^4}={5\over32\Om^7}.
\label{eq:p-integrals}
\end{equation}
Therefore
\begin{align}
G_E(\tau,\tau)
&={1\over2\Om}-{V''\over16\Om^5}+{5(V')^2\over64\Om^7}+\Oh(\partial_\tau^4) \nonumber\\
&={1\over2\Om}-{\ddot\Om\over8\Om^4}+{3\dot\Om^2\over16\Om^5}+\Oh(\partial_\tau^4),
\label{eq:GE-diagonal}
\end{align}
where \(V'=2\Om\dot\Om\) and \(V''=2\dot\Om^2+2\Om\ddot\Om\) were used.

The determinant obeys
\begin{equation}
\delta\Delta\Gamma_E^{(1)}={\hbar\over2}\Tr(A^{-1}\delta A)
={\hbar\over2}\int\dd\tau\,G_E(\tau,\tau)\delta\Om^2(\tau).
\label{eq:var-det}
\end{equation}
To reconstruct a local Lagrangian, set
\begin{equation}
L_1(\Om,\dot\Om)={1\over2}\Om+a{\dot\Om^2\over\Om^3}.
\label{eq:L1-ansatz}
\end{equation}
Since \(\delta\Om=\delta(\Om^2)/(2\Om)\),
\begin{align}
{\delta\over\delta\Om^2}\int\dd\tau\,L_1
&={1\over2\Om}\left({\partial L_1\over\partial\Om}-{\dd\over\dd\tau}{\partial L_1\over\partial\dot\Om}\right) \nonumber\\
&={1\over4\Om}-a{\ddot\Om\over\Om^4}+{3a\over2}{\dot\Om^2\over\Om^5}.
\label{eq:func-der-L1}
\end{align}
Comparison with \(G_E/2\) from Eq.~\eqref{eq:GE-diagonal} fixes \(a=1/16\).  Hence
\begin{equation}
\Delta\Gamma_E^{(1)}
=\hbar\int\dd\tau\left[{1\over2}(\Om-\omega)+{\dot\Om^2\over16\Om^3}+\Oh(\partial_\tau^4)\right].
\label{eq:det-derivative-expanded}
\end{equation}
Using Eq.~\eqref{eq:operator},
\begin{equation}
2\Om\dot\Om={\lambda\over m}Q\dot Q,
\qquad
\dot\Om={\lambda Q\dot Q\over2m\Om},
\label{eq:Omdot}
\end{equation}
and therefore
\begin{equation}
{\dot\Om^2\over16\Om^3}={\lambda^2Q^2\dot Q^2\over64m^2\Om^5}.
\label{eq:Omdot-term}
\end{equation}
The local Euclidean effective action is then
\begin{equation}
\Gamma_E[Q]\simeq\int\dd\tau\left[{1\over2}Z(Q)\dot Q^2+V_{\rm eff}(Q)\right],
\label{eq:Euclidean-local-action}
\end{equation}
with
\begin{equation}
V_{\rm eff}(Q)={m\omega^2\over2}Q^2+{\lambda\over24}Q^4+{\hbar\over2}\big[\Om(Q)-\omega\big],
\qquad
\Om(Q)=\sqrt{\omega^2+{\lambda\over2m}Q^2},
\label{eq:Veff}
\end{equation}
\begin{equation}
Z(Q)=m+\hbar{\lambda^2Q^2\over32m^2\Om^5}.
\label{eq:ZQ}
\end{equation}

\section{Local equation and Wick rotation}

For
\begin{equation}
L_E={1\over2}Z(Q)\,\dot Q^2+V_{\rm eff}(Q),
\label{eq:LE}
\end{equation}
one has
\begin{equation}
{\partial L_E\over\partial Q}={1\over2}Z'(Q)\dot Q^2+V'_{\rm eff}(Q),
\qquad
{\partial L_E\over\partial\dot Q}=Z(Q)\,\dot Q,
\label{eq:EL-pieces-1}
\end{equation}
\begin{equation}
{\dd\over\dd\tau}{\partial L_E\over\partial\dot Q}=Z(Q)\ddot Q+Z'(Q)\dot Q^2.
\label{eq:EL-pieces-2}
\end{equation}
Thus \(\partial L_E/\partial Q-\dd(\partial L_E/\partial\dot Q)/\dd\tau=0\) gives
\begin{equation}
-Z(Q)\ddot Q-{1\over2}Z'(Q)\dot Q^2+V'_{\rm eff}(Q)=0.
\label{eq:E-eom-local}
\end{equation}
The derivative of \(\Om\) with respect to \(Q\) is
\begin{equation}
\Om'(Q)={\lambda Q\over2m\Om}.
\label{eq:Omprime}
\end{equation}
Therefore
\begin{equation}
V'_{\rm eff}(Q)=m\omega^2Q+{\lambda\over6}Q^3+{\hbar\over2}\Om'(Q)
=m\omega^2Q+{\lambda\over6}Q^3+\hbar{\lambda Q\over4m\Om}.
\label{eq:Veffprime}
\end{equation}
Also,
\begin{align}
Z'(Q)
&={\hbar\lambda^2\over32m^2}{\dd\over\dd Q}\left(Q^2\Om^{-5}\right) \nonumber\\
&={\hbar\lambda^2\over32m^2}\left[2Q\Om^{-5}-5Q^2\Om^{-6}\Om'(Q)\right] \nonumber\\
&={\hbar\lambda^2\over32m^2}\left[{2Q\over\Om^5}-{5\lambda Q^3\over2m\Om^7}\right].
\label{eq:Zprime}
\end{align}
Now rotate the already local equation by
\begin{equation}
\tau=\ii t,
\qquad
{\dd Q\over\dd\tau}=-\ii {\dd Q\over\dd t},
\qquad
{\dd^2Q\over\dd\tau^2}=-{\dd^2Q\over\dd t^2}.
\label{eq:wick}
\end{equation}
Equation \eqref{eq:E-eom-local} becomes
\begin{equation}
Z(Q)\ddot Q+{1\over2}Z'(Q)\dot Q^2+V'_{\rm eff}(Q)=0,
\label{eq:RT-local-general}
\end{equation}
where dots now denote \(\dd/\dd t\).  Substituting Eqs.~\eqref{eq:ZQ}, \eqref{eq:Veffprime}, and \eqref{eq:Zprime}, the local adiabatic tree plus one-loop equation is
\begin{equation}
\boxed{\begin{aligned}
0={}&\left[m+\hbar{\lambda^2Q^2\over32m^2\Om^5}\right]\ddot Q
+{\hbar\lambda^2\over64m^2}\left[{2Q\over\Om^5}-{5\lambda Q^3\over2m\Om^7}\right]\dot Q^2 \\
&+m\omega^2Q+{\lambda\over6}Q^3+\hbar{\lambda Q\over4m\Om},
\qquad
\Om=\sqrt{\omega^2+{\lambda\over2m}Q^2}.
\end{aligned}}
\tag{A}
\label{eq:final-adiabatic}
\end{equation}
The derivative expansion assumes
\begin{equation}
{|\dot\Om|\over\Om^2}\ll1,
\qquad
{|\ddot\Om|\over\Om^3}\ll1,
\label{eq:adiabatic-condition}
\end{equation}
and omits \(\Oh(\hbar\partial_t^4)\) and \(\Oh(\hbar^2)\) terms.  Equation \eqref{eq:final-adiabatic} is conservative because it follows from
\begin{equation}
\Gamma_{\rm RT}^{\rm loc}=\int\dd t\left[{1\over2}Z(Q)\dot Q^2-V_{\rm eff}(Q)\right].
\label{eq:RT-loc-action}
\end{equation}

\section{Real-time correlator from mode averaging}

The Heisenberg equation is
\begin{equation}
m\ddot q+m\omega^2q+{\lambda\over6}q^3=0.
\label{eq:heisenberg}
\end{equation}
Write \(q=Q+\eta\) with \(Q=\langle q\rangle\), \(\langle\eta\rangle=0\).  Since
\begin{equation}
\langle(Q+\eta)^3\rangle=Q^3+3Q\langle\eta^2\rangle+\langle\eta^3\rangle,
\label{eq:cubic-average}
\end{equation}
and \(\langle\eta^3\rangle=0\) in the Gaussian one-loop approximation, the one-loop tadpole equation is
\begin{equation}
0=m\ddot Q+m\omega^2Q+{\lambda\over6}Q^3+{\lambda\over2}Q(t)\langle\eta^2(t)\rangle.
\label{eq:tadpole-eom}
\end{equation}
The quadratic real-time action for \(\eta\) is
\begin{equation}
S_2[\eta;Q]={m\over2}\int\dd t\left[\dot\eta^2-\Om_Q^2(t)\eta^2\right],
\qquad
\Om_Q^2(t)=\omega^2+\delta\Om^2(t),
\qquad
\delta\Om^2(t)={\lambda\over2m}Q^2(t).
\label{eq:S2-real}
\end{equation}
Thus each fluctuation mode obeys the oscillator equation with time-dependent frequency.  For the single quantum-mechanical oscillator there is one mode; in field theory the replacement is a sum/integral over momenta.  Use
\begin{equation}
\hat\eta(t)=\sqrt{\hbar\over2m}\left[a u(t)+a^\dagger u^*(t)\right],
\qquad
[a,a^\dagger]=1.
\label{eq:mode-expansion}
\end{equation}
The equation of motion and Wronskian normalization are
\begin{equation}
\ddot u+\Om_Q^2(t)u=0,
\qquad
W[u]\equiv u\dot u^*-\dot u u^*=2\ii.
\label{eq:mode-eq-wronskian}
\end{equation}
The Wronskian gives the canonical commutator because
\begin{equation}
[\hat\eta,m\dot{\hat\eta}]={\hbar\over2}\left(u\dot u^*-u^*\dot u\right)=\ii\hbar.
\label{eq:commutator-check}
\end{equation}
In the state annihilated by \(a\),
\begin{equation}
\langle\eta^2(t)\rangle={\hbar\over2m}|u(t)|^2.
\label{eq:eta2-exact-mode}
\end{equation}
Equations \eqref{eq:tadpole-eom}, \eqref{eq:mode-eq-wronskian}, and \eqref{eq:eta2-exact-mode} give the exact causal one-loop backreaction system.  The next steps compute its first explicit nonlocal correction.

Choose the free positive-frequency initial mode
\begin{equation}
u_0(t)={e^{-\ii\omega(t-t_0)}\over\sqrt\omega},
\qquad
\ddot u_0+\omega^2u_0=0,
\qquad
u_0\dot u_0^*-\dot u_0u_0^*=2\ii.
\label{eq:u0}
\end{equation}
Write \(u=u_0+\delta u\).  To first order in \(\delta\Om^2\),
\begin{equation}
(\partial_t^2+\omega^2)\delta u(t)=-\delta\Om^2(t)u_0(t).
\label{eq:deltau-eq}
\end{equation}
The retarded Green function of \(\partial_t^2+\omega^2\) is
\begin{equation}
G_R(t-t')=\theta(t-t'){\sin[\omega(t-t')]\over\omega},
\qquad
(\partial_t^2+\omega^2)G_R(t-t')=\delta(t-t').
\label{eq:GR-osc}
\end{equation}
Therefore
\begin{equation}
\delta u(t)=-\int_{t_0}^{t}\dd t'\,{\sin[\omega(t-t')]\over\omega}\,\delta\Om^2(t')u_0(t').
\label{eq:deltau-solution}
\end{equation}
Let \(\Delta=t-t'\).  Then
\begin{align}
u_0^*(t)\delta u(t)
&=-\int_{t_0}^{t}\dd t'\,{\sin(\omega\Delta)\over\omega^2}e^{\ii\omega\Delta}\delta\Om^2(t'),
\label{eq:u0star-deltau}
\end{align}
and adding the complex conjugate gives
\begin{align}
|u(t)|^2
&=|u_0|^2+u_0^*\delta u+u_0\delta u^*+\Oh((\delta\Om^2)^2) \nonumber\\
&={1\over\omega}-{2\over\omega^2}\int_{t_0}^{t}\dd t'\,\sin(\omega\Delta)\cos(\omega\Delta)\delta\Om^2(t')+\Oh((\delta\Om^2)^2) \nonumber\\
&={1\over\omega}-{1\over\omega^2}\int_{t_0}^{t}\dd t'\,\sin[2\omega(t-t')]\delta\Om^2(t')+\Oh((\delta\Om^2)^2).
\label{eq:u-square}
\end{align}
Using \(\delta\Om^2=\lambda Q^2/(2m)\),
\begin{equation}
\langle\eta^2(t)\rangle={\hbar\over2m\omega}
-{\hbar\lambda\over4m^2\omega^2}\int_{t_0}^{t}\dd t'\,\sin[2\omega(t-t')]Q^2(t')
+\Oh\left({\hbar\over m}{(\delta\Om^2)^2\over\omega^5}\right).
\label{eq:eta2-memory}
\end{equation}
Substitution into Eq.~\eqref{eq:tadpole-eom} gives the explicit causal nonlocal tree plus one-loop equation
\begin{equation}
\boxed{\begin{aligned}
0={}&m\ddot Q(t)+m\omega^2Q(t)+{\lambda\over6}Q^3(t)
+\hbar{\lambda\over4m\omega}Q(t) \\
&-\hbar{\lambda^2\over8m^2\omega^2}Q(t)
\int_{t_0}^{t}\dd t'\,\sin[2\omega(t-t')]Q^2(t')
+\Oh\left({\hbar\lambda Q\over m}{(\delta\Om^2)^2\over\omega^5}\right).
\end{aligned}}
\tag{B}
\label{eq:final-nonlocal}
\end{equation}
For finite \(t_0\), Eq.~\eqref{eq:final-nonlocal} contains physical initial-time transients.  To compare with the adiabatic effective action one takes the usual retarded adiabatic-switching limit, equivalently \(t_0\to-\infty\) with a factor \(e^{-\epsilon(t-t')}\), \(\epsilon\to0^+\), in the memory integral.

\section{Local limit and coefficient check}

Define the adiabatically switched memory functional
\begin{equation}
I[f](t)=\lim_{\epsilon\to0^+}\int_0^\infty\dd s\,e^{-\epsilon s}\sin(2\omega s)f(t-s),
\qquad
f(t)=Q^2(t).
\label{eq:I-def}
\end{equation}
For slowly varying \(f\),
\begin{equation}
f(t-s)=f(t)-s\dot f(t)+{s^2\over2}\ddot f(t)+\cdots.
\label{eq:f-expand}
\end{equation}
The regulated moments are
\begin{equation}
\int_0^\infty\dd s\,e^{-\epsilon s}\sin(as)={a\over a^2+\epsilon^2}\xrightarrow{\epsilon\to0^+}{1\over a},
\label{eq:moment0}
\end{equation}
\begin{equation}
\int_0^\infty\dd s\,s e^{-\epsilon s}\sin(as)={2a\epsilon\over(a^2+\epsilon^2)^2}\xrightarrow{\epsilon\to0^+}0,
\label{eq:moment1}
\end{equation}
\begin{equation}
\int_0^\infty\dd s\,s^2 e^{-\epsilon s}\sin(as)
={2a(3\epsilon^2-a^2)\over(a^2+\epsilon^2)^3}
\xrightarrow{\epsilon\to0^+}-{2\over a^3}.
\label{eq:moment2}
\end{equation}
With \(a=2\omega\),
\begin{equation}
I[Q^2](t)={Q^2\over2\omega}-{1\over(2\omega)^3}{\dd^2Q^2\over\dd t^2}+\Oh(\partial_t^4).
\label{eq:I-local}
\end{equation}
The memory force in Eq.~\eqref{eq:final-nonlocal} therefore becomes
\begin{align}
-\hbar{\lambda^2\over8m^2\omega^2}Q I[Q^2]
&=-\hbar{\lambda^2Q^3\over16m^2\omega^3}
+\hbar{\lambda^2\over64m^2\omega^5}Q{\dd^2Q^2\over\dd t^2}+\Oh(\partial_t^4) \nonumber\\
&=-\hbar{\lambda^2Q^3\over16m^2\omega^3}
+\hbar{\lambda^2\over32m^2\omega^5}\left(Q^2\ddot Q+Q\dot Q^2\right)+\Oh(\partial_t^4),
\label{eq:memory-local-force}
\end{align}
because
\begin{equation}
Q{\dd^2Q^2\over\dd t^2}=Q{\dd\over\dd t}(2Q\dot Q)=2Q^2\ddot Q+2Q\dot Q^2.
\label{eq:Q2ddot}
\end{equation}
Now expand the adiabatic equation \eqref{eq:final-adiabatic} to the same order.  Since
\begin{equation}
{1\over\Om}={1\over\omega}\left(1-{\lambda Q^2\over4m\omega^2}+\Oh(\lambda^2Q^4)\right),
\qquad
{1\over\Om^5}={1\over\omega^5}+\Oh(\lambda Q^2),
\label{eq:Omega-smallQ}
\end{equation}
Eq.~\eqref{eq:final-adiabatic} gives, through \(\Oh(\hbar\lambda^2)\),
\begin{equation}
0=m\ddot Q+m\omega^2Q+{\lambda\over6}Q^3
+\hbar{\lambda Q\over4m\omega}
-\hbar{\lambda^2Q^3\over16m^2\omega^3}
+\hbar{\lambda^2\over32m^2\omega^5}\left(Q^2\ddot Q+Q\dot Q^2\right)+\cdots.
\label{eq:adiabatic-smallQ}
\end{equation}
This is exactly the local derivative expansion of Eq.~\eqref{eq:final-nonlocal} in the adiabatic-switching limit.  The recheck also shows why the \(\dot Q^2\) term in Eq.~\eqref{eq:final-adiabatic} is not friction: it is part of the conservative local expansion of the causal memory kernel.
