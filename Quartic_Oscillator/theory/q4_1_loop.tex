{\color{Navy}\LARGE\bfseries Quartic Oscillator at One Loop}\\[-1pt]
{\large\color{Teal}Local adiabatic dynamics and causal nonlocal memory}\\[2pt]
\end{center}
\vspace{-2pt}

\begin{summarybox}[Scope and results]
For
$S[q]=\int\dd t\,[m\dot q^2/2-m\omega^2q^2/2-\lambda q^4/24]$,
this note derives two complementary one-loop equations.  Equation~\eqref{eq:A} is local, second adiabatic order, and nonperturbative in the background amplitude.  Equation~\eqref{eq:B} is causal and retarded, but first Born in $\delta\Om^2=\lambda Q^2/(2m)$.  Their coefficients agree in the common weak-background, low-frequency, adiabatically prepared regime.  Every remainder and state assumption used below is stated explicitly.
\end{summarybox}

\section{Model, assumptions, and power counting}

The Euclidean action and local fluctuation frequency are
\begin{equation}
S_E[q]=\int\dd\tau\left[\frac m2\dot q^2+\frac{m\omega^2}{2}q^2+\frac\lambda{24}q^4\right],
\qquad
\Om^2(Q)=\omega^2+\frac{\lambda Q^2}{2m}.
\label{eq:model}
\end{equation}
We assume $m>0$, $\omega>0$, $\lambda\ge0$, and $\Om\ge\Om_{\min}>0$.  Euclidean time is the zero-temperature infinite line; integrations by parts are legitimate, and the background and reference determinants use identical boundary conditions.  ``One loop'' means terms through $O(\hbar)$; a self-consistent solution of a truncated equation may resum uncontrolled higher powers of $\hbar$, so strict loop accuracy requires perturbative solution or order reduction.

\section{Euclidean one-loop action}

\subsection{Background expansion and determinant}

Set $q=Q+\eta$.  Using $\int m\dot Q\dot\eta\,\dd\tau=-\int m\ddot Q\eta\,\dd\tau$ and the binomial expansion,
\begin{align}
S_E[Q+\eta]={}&S_E[Q]+\int\dd\tau\,\eta E_Q
+\frac12\int\dd\tau\,\eta\mathcal O_Q\eta
+\int\dd\tau\left(\frac\lambda6Q\eta^3+\frac\lambda{24}\eta^4\right),
\label{eq:expanded}\\[1mm]
E_Q={}&-m\ddot Q+m\omega^2Q+\frac\lambda6Q^3,
\qquad
\mathcal O_Q=m\left[-\partial_\tau^2+\Om^2(\tau)\right].
\label{eq:E-and-O}
\end{align}
The effective-action source cancels the linear term.  Cubic and quartic fluctuation vertices start at two loops, hence
\begin{equation}
\Gamma_E[Q]=S_E[Q]+\frac\hbar2\Tr\ln\mathcal O_Q+O(\hbar^2).
\label{eq:Gamma1}
\end{equation}
After removing the field-independent factor $m$ and normalizing at $Q=0$,
\begin{equation}
\Delta\Gamma_E^{(1)}=\frac\hbar2\Tr\ln[-\partial_\tau^2+\Om^2]
-\frac\hbar2\Tr\ln[-\partial_\tau^2+\omega^2].
\label{eq:det}
\end{equation}

\subsection{Weyl inverse through second adiabatic order}

Let $A=-\partial_\tau^2+V(\tau)$, $V=\Om^2$, $f=p^2+V$, and let $R$ be the Weyl symbol of $A^{-1}$.  The specialized Moyal equation $f\star R=1$ is
\begin{equation}
f\star R=fR+\frac\ii2(V'R_p-2pR_\tau)
-\frac18(V''R_{pp}+2R_{\tau\tau})
-\frac\ii{48}V'''R_{ppp}+O(\partial_\tau^4).
\label{eq:moyal}
\end{equation}
With $R=R_0+R_1+R_2+R_3+\cdots$,
\begin{equation}
R_0=\frac1f,
\qquad R_1=0,
\qquad
R_2=-\frac{V''}{2f^3}+\frac{V''p^2}{f^4}+\frac{(V')^2}{2f^4}.
\label{eq:R012}
\end{equation}
The third-order terms cancel rather than being absent by assumption:
\begin{equation}
\{f,R_2\}_{\rm PB}=-\frac{V'''p(p^2-V)}{(p^2+V)^4}
=\frac{V'''}{24}(R_0)_{ppp}
\quad\Longrightarrow\quad R_3=0.
\label{eq:R3}
\end{equation}
Thus $R=R_0+R_2+O(\partial_\tau^4)$.  The master integral
\begin{equation}
\int_{-\infty}^{\infty}\frac{\dd p}{2\pi}\frac1{(p^2+\Om^2)^n}
=\frac{\Gamma(n-\tfrac12)}{2\sqrt\pi\,\Gamma(n)}\Om^{1-2n},\qquad n>\tfrac12
\label{eq:master}
\end{equation}
and $p^2/f^4=f^{-3}-\Om^2f^{-4}$ give
\begin{equation}
\GE(\tau,\tau)=\int\frac{\dd p}{2\pi}R
=\frac1{2\Om}-\frac{\ddot\Om}{8\Om^4}+\frac{3\dot\Om^2}{16\Om^5}
+O(\partial_\tau^4).
\label{eq:GE}
\end{equation}

\subsection{Local determinant and effective coefficients}

Because
\begin{equation}
\delta\Delta\Gamma_E^{(1)}=\frac\hbar2\int\dd\tau\,\GE(\tau,\tau)\,\delta\Om^2,
\label{eq:varDet}
\end{equation}
write, up to a total derivative,
\begin{equation}
L_1=\frac12\Om+a\frac{\dot\Om^2}{\Om^3},
\qquad
\frac{\delta}{\delta\Om^2}\int L_1\dd\tau
=\frac1{4\Om}-a\frac{\ddot\Om}{\Om^4}+\frac{3a}{2}\frac{\dot\Om^2}{\Om^5}.
\label{eq:L1}
\end{equation}
Matching Eq.~\eqref{eq:GE}/2 fixes $a=1/16$.  Since $2\Om\dot\Om=(\lambda/m)Q\dot Q$,
\begin{equation}
\Gamma_E^{\rm loc}=\int\dd\tau\left[\frac12Z(Q)\dot Q^2+V_{\rm eff}(Q)\right]
+O(\hbar\partial_\tau^4)+O(\hbar^2),
\label{eq:localE}
\end{equation}
with
\begin{equation}
V_{\rm eff}=\frac{m\omega^2}{2}Q^2+\frac\lambda{24}Q^4+\frac\hbar2(\Om-\omega),
\qquad
Z(Q)=m+\frac{\hbar\lambda^2Q^2}{32m^2\Om^5}.
\label{eq:VZ}
\end{equation}

\section{Local real-time equation}

Under $\tau=\ii t$, the local action becomes
\begin{equation}
\Gamma_{\rm RT}^{\rm loc}=\int\dd t\left[\frac12Z(Q)\dot Q^2-V_{\rm eff}(Q)\right],
\qquad
Z\ddot Q+\frac12Z'\dot Q^2+V'_{\rm eff}=0.
\label{eq:RTA}
\end{equation}
Here
\begin{equation}
\Om'=\frac{\lambda Q}{2m\Om},
\quad
V'_{\rm eff}=m\omega^2Q+\frac\lambda6Q^3+\frac{\hbar\lambda Q}{4m\Om},
\quad
Z'=\frac{\hbar\lambda^2}{32m^2}\left(\frac{2Q}{\Om^5}-\frac{5\lambda Q^3}{2m\Om^7}\right).
\label{eq:derivs}
\end{equation}

\begin{resultbox}{Result A - local adiabatic equation}
\begingroup\renewcommand{\theHequation}{A}
\begin{equation}
\begin{aligned}
0={}&\left[m+\frac{\hbar\lambda^2Q^2}{32m^2\Om^5}\right]\ddot Q
+\frac{\hbar\lambda^2}{64m^2}
\left(\frac{2Q}{\Om^5}-\frac{5\lambda Q^3}{2m\Om^7}\right)\dot Q^2\\
&+m\omega^2Q+\frac\lambda6Q^3+\frac{\hbar\lambda Q}{4m\Om},
\qquad
\Om=\sqrt{\omega^2+\frac{\lambda Q^2}{2m}}.
\end{aligned}
\tag{A}\label{eq:A}
\end{equation}
\endgroup
\end{resultbox}

The truncated energy
\begin{equation}
E_{\rm loc}=\frac12Z\dot Q^2+V_{\rm eff},
\qquad
\dot E_{\rm loc}=\dot Q\left(Z\ddot Q+\tfrac12Z'\dot Q^2+V'_{\rm eff}\right)=0,
\label{eq:Eloc}
\end{equation}
shows that the $\dot Q^2$ force is not friction.

\begin{validitybox}[Validity of Result A]
A controlled derivative expansion requires a slow family $\Om_\epsilon(t)=\bar\Om(\epsilon t)$ with
$\Om_\epsilon^{(n)}/\Om_\epsilon^{n+1}=O[(\epsilon/\Om_{\min})^n]$ for all relevant $n$, not merely two pointwise inequalities.  Analytic continuation of the Euclidean 1PI functional is an in--out construction; its causal meaning is established below by the in--in memory limit and by an independent adiabatic-mode check.  Result A omits $O(\hbar\partial_t^4)$ and $O(\hbar^2)$.
\end{validitybox}

\section{Causal Gaussian dynamics and first-Born memory}

The Heisenberg equation and $q=Q+\eta$, $Q=\langle q\rangle$, $\langle\eta\rangle=0$ give
\begin{equation}
0=m\ddot Q+m\omega^2Q+\frac\lambda6Q^3+\frac\lambda2Q\langle\eta^2\rangle_0+\Dtwo,
\qquad \Dtwo\text{ collects two-loop forces},
\label{eq:mean}
\end{equation}
because $\langle\eta^3\rangle$ starts beyond one loop.  For a pure Gaussian state,
\begin{gather}
\hat\eta=\sqrt{\frac\hbar{2m}}\,[a u+a^\dagger u^*],
\qquad
\ddot u+\Om_Q^2u=0,
\qquad
W[u]\equiv u\dot u^*-\dot u\,u^*=2\ii,
\label{eq:mode}\\[-1mm]
\Om_Q^2=\omega^2+\delta\Om^2,
\qquad
\delta\Om^2=\frac{\lambda Q^2}{2m},
\qquad
\langle\eta^2\rangle_0=\frac\hbar{2m}|u|^2.
\label{eq:variance}
\end{gather}
The Wronskian gives $[\hat\eta,m\dot{\hat\eta}]=\ii\hbar$.  With $\Dtwo$ omitted, the closed Gaussian system conserves
\begin{equation}
E_G=\frac m2\dot Q^2+\frac{m\omega^2}{2}Q^2+\frac\lambda{24}Q^4
+\frac\hbar4\left(|\dot u|^2+\Om_Q^2|u|^2\right).
\label{eq:EG}
\end{equation}
This is reversible mean--fluctuation energy exchange, not automatic dissipation.

Choose the free-vacuum initial mode
\begin{equation}
u_0(t)=\frac{e^{-\ii\omega(t-t_0)}}{\sqrt\omega},
\qquad
\delta u(t_0)=\delta\dot u(t_0)=0.
\label{eq:u0}
\end{equation}
At first Born order,
\begin{equation}
\delta u_1(t)=-\int_{t_0}^{t}\dd t'\,
\frac{\sin[\omega(t-t')]}{\omega}\,\delta\Om^2(t')u_0(t'),
\label{eq:du}
\end{equation}
so
\begin{equation}
|u(t)|^2=\frac1\omega-\frac1{\omega^2}\int_{t_0}^{t}\dd t'\,
\sin[2\omega(t-t')]\delta\Om^2(t')+R_u^{(2)}(t).
\label{eq:u2Born}
\end{equation}
A dimensionally correct finite-time control follows from
\begin{equation}
\kappa_T=\sup_{t_0\le t\le T}\frac1\omega\int_{t_0}^{t}\dd t'\,|\delta\Om^2(t')|.
\label{eq:kappa}
\end{equation}
For $\kappa_T<1$, the Volterra--Neumann series converges and one may bound
\begin{equation}
|R_u^{(2)}(t)|\le
\frac{\kappa_T^2(3-2\kappa_T)}{\omega(1-\kappa_T)^2},
\qquad t\in[t_0,T].
\label{eq:Rbound}
\end{equation}
Thus pointwise $|\delta\Om^2|/\omega^2\ll1$ alone is not a uniform long-time guarantee.

Using Eq.~\eqref{eq:variance} in Eq.~\eqref{eq:mean} gives the retarded equation below, with
$\DBorn=(\hbar\lambda Q/4m)R_u^{(2)}$.

\begin{resultbox}{Result B - causal first-Born memory equation}
\begingroup\renewcommand{\theHequation}{B}
\begin{equation}
\begin{aligned}
0={}&m\ddot Q(t)+m\omega^2Q(t)+\frac\lambda6Q^3(t)
+\frac{\hbar\lambda}{4m\omega}Q(t)\\
&-\frac{\hbar\lambda^2}{8m^2\omega^2}Q(t)
\int_{t_0}^{t}\dd t'\,\sin[2\omega(t-t')]Q^2(t')
+\DBorn(t)+\Dtwo(t).
\end{aligned}
\tag{B}\label{eq:B}
\end{equation}
\endgroup
\end{resultbox}

Finite-$t_0$ oscillations are state-preparation transients.  They describe a real quench when that is the preparation, but are artifacts if the intended state is the interacting adiabatic vacuum.  For comparison with the Euclidean vacuum functional, take $t_0\to-\infty$ with $e^{-\epsilon(t-t')}$ inserted before $\epsilon\to0^+$.

\section{Local limit and independent coefficient checks}

\subsection{Low-frequency expansion of the memory kernel}

With $a=2\omega$, define
\begin{equation}
I[f](t)=\lim_{\epsilon\to0^+}\int_0^\infty\dd s\,e^{-\epsilon s}\sin(as)f(t-s).
\label{eq:I}
\end{equation}
For $f(t)=\int(\dd\nu/2\pi)e^{-\ii\nu t}\tilde f(\nu)$,
\begin{equation}
\widetilde{I[f]}(\nu)=\frac{a}{a^2-(\nu+\ii0)^2}\tilde f(\nu).
\label{eq:transfer}
\end{equation}
If the spectrum is confined well inside $|\nu|<a$, then
\begin{equation}
I[f]=\frac f a-\frac{\ddot f}{a^3}+\frac{f^{(4)}}{a^5}+\cdots.
\label{eq:Iexp}
\end{equation}
The nearest poles are at $\nu=\pm2\omega$; no local series is controlled near that pair resonance.  For $f=Q^2$,
\begin{align}
-\frac{\hbar\lambda^2}{8m^2\omega^2}QI[Q^2]
={}&-\frac{\hbar\lambda^2Q^3}{16m^2\omega^3}
+\frac{\hbar\lambda^2}{32m^2\omega^5}
\left(Q^2\ddot Q+Q\dot Q^2\right)
+O(\hbar\partial_t^4).
\label{eq:memorylocal}
\end{align}

Introduce the dimensionless amplitude
\begin{equation}
x=\frac{\lambda Q^2}{2m\omega^2},
\qquad
\frac1\Om=\frac1\omega\left(1-\frac x2+O(x^2)\right),
\qquad
\frac1{\Om^5}=\frac1{\omega^5}\left(1-\frac{5x}{2}+O(x^2)\right).
\label{eq:x}
\end{equation}
Expanding Result A through $O(\hbar\lambda^2)$ and second derivative order gives
\begin{equation}
\begin{aligned}
0={}&m\ddot Q+m\omega^2Q+\frac\lambda6Q^3+\frac{\hbar\lambda Q}{4m\omega}
-\frac{\hbar\lambda^2Q^3}{16m^2\omega^3}\\
&+\frac{\hbar\lambda^2}{32m^2\omega^5}
\left(Q^2\ddot Q+Q\dot Q^2\right)+\text{higher orders},
\end{aligned}
\label{eq:match}
\end{equation}
which exactly matches the displayed local expansion of Result B at that order.

\subsection{Real-time adiabatic/WKB check}

The stronger check does not expand in $x$.  For the adiabatic vacuum, take
\begin{equation}
u(t)=\frac1{\sqrt{W(t)}}\exp\left[-\ii\int^tW(t')\dd t'\right],
\qquad
W^2=\Om^2-\frac{\ddot W}{2W}+\frac{3\dot W^2}{4W^2}.
\label{eq:WKB}
\end{equation}
Iterating through second adiabatic order,
\begin{equation}
|u|^2=\frac1\Om+\frac{\ddot\Om}{4\Om^4}-\frac{3\dot\Om^2}{8\Om^5}+O(\partial_t^4).
\label{eq:uWKB}
\end{equation}
Using
\begin{equation}
\dot\Om=\frac{\lambda Q\dot Q}{2m\Om},
\qquad
\ddot\Om=\frac{\lambda(\dot Q^2+Q\ddot Q)}{2m\Om}
-\frac{\lambda^2Q^2\dot Q^2}{4m^2\Om^3},
\label{eq:Omdots}
\end{equation}
the tadpole force becomes
\begin{align}
\frac\lambda2Q\langle\eta^2\rangle_0
={}&\frac{\hbar\lambda Q}{4m\Om}
+\frac{\hbar\lambda^2Q^2}{32m^2\Om^5}\ddot Q
+\frac{\hbar\lambda^2}{64m^2}
\left(\frac{2Q}{\Om^5}-\frac{5\lambda Q^3}{2m\Om^7}\right)\dot Q^2
+O(\hbar\partial_t^4),
\label{eq:WKBmatch}
\end{align}
reproducing every displayed one-loop coefficient in Result A.
